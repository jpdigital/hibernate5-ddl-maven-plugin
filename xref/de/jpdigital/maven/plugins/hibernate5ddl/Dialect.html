89      /**
90       * Property for holding the name of the Hibernate dialect class.
91       */
92      private final String dialectClass;
93  
94      /**
95       * Private constructor, used to create the Enum instances for each dialect.
96       *
97       * @param dialectClass The dialect class for the specific dialect.
98       */
99      private Dialect(final String dialectClass) {
100         this.dialectClass = dialectClass;
101     }
102 
103     /**
104      * Getter for the dialect class.
105      *
106      * @return The name of the dialect class, for example
107      *         {@code org.hibernate.dialect.PostgreSQL9Dialect} for
108      *         {@link #POSTGRESQL9}.
109      */
110     public String getDialectClass() {
111         return dialectClass;
