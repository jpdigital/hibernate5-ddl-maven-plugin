116     /**
117      * Property for holding the name of the Hibernate dialect class.
118      */
119     private final String dialectClassName;
120 
121     /**
122      * Private constructor, used to create the Enum instances for each dialect.
123      *
124      * @param dialectClass The dialect class for the specific dialect.
125      */
126     private Dialect(final String dialectClass) {
127         this.dialectClassName = dialectClass;
128     }
129 
130     /**
131      * Getter for the dialect class.
132      *
133      * @return The name of the dialect class, for example
134      *         {@code org.hibernate.dialect.PostgreSQL9Dialect} for
135      *         {@link #POSTGRESQL9}.
136      */
137     public String getDialectClassName() {
138         return dialectClassName;
139     }
