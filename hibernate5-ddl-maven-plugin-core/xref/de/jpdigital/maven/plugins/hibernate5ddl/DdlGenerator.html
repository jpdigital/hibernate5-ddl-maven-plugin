17  package de.jpdigital.maven.plugins.hibernate5ddl;
18  
19  import org.apache.maven.plugin.MojoFailureException;
20  
21  import java.util.ServiceLoader;
22  import java.util.Set;
23  
24  /**
25   * Interface for the {@code DdlGenerator} which generates the SQL DDL files.
26   *
27   * The {@link GenerateDdlMojo} will use the {@link ServiceLoader} from the Java
28   * Standard API to find the implementation to use. Therefore an implementation
29   * of this interface must be accompanied by a file called
30   * {@code de.jpdigital.maven.plugins.hibernate5ddl.DdlGenerator} in the
31   * {@code META-INF/services} directory.
32   *
33   * @author <a href="mailto:jens.pelzetter@googlemail.com">Jens Pelzetter</a>
34   */
35  public interface DdlGenerator {
36  
37      /**
38       * Generates a SQL DDL file for a specific SQL dialect.
39       *
40       * @param dialect       The SQL dialect to use.
41       * @param entityClasses The entity classes for which SQL DDL statements will
42       *                      be created.
43       * @param mojo          The {@link GenerateDdlMojo} which calls the method.
44       *
45       * @throws MojoFailureException If an error occurs while creating the DDL
46       *                              file.
47       *
48       * @see Dialect
49       */
50      void generateDdl(Dialect dialect,
51                       Set<Class<?>> entityClasses,
52                       GenerateDdlMojo mojo)
53          throws MojoFailureException;
54  
55      /**
56       * Generates a SQL DDL file for a specific SQL dialect. This method accepts
57       * a string a is called by {@link GenerateDdlMojo} for processing custom
58       * dialects.
59       *
60       * @param dialect       The SQL dialect to use.
61       * @param entityClasses The entity classes for which SQL DDL statements will
62       *                      be created.
63       * @param mojo          The {@link GenerateDdlMojo} which calls the method.
64       *
65       * @throws MojoFailureException If an error occurs while creating the DDL
66       *                              file.
67       *
68       */
69      void generateDdl(String dialect,
70                       Set<Class<?>> entityClasses,
71                       GenerateDdlMojo mojo)
72          throws MojoFailureException;
73  
74  }
