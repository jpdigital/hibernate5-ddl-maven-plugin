137     /**
138      * The Mojo's execute method.
139      *
140      * @throws MojoExecutionException if the Mojo can't be executed.
141      * @throws MojoFailureException   if something goes wrong while to Mojo is
142      *                                executed.
143      */
144     @Override
145     @SuppressWarnings({"PMD.CyclomaticComplexity"})
146     public void execute() throws MojoExecutionException, MojoFailureException {
147         final File outputDir = outputDirectory;
148 
149         getLog().info(String.format("Generating DDL SQL files in %s.",
150                                     outputDir.getAbsolutePath()));
151 
152         //Check if the output directory exists.
153         if (!outputDir.exists()) {
154             final boolean result = outputDir.mkdirs();
155             if (!result) {
156                 throw new MojoFailureException(
157                     "Failed to create output directory for SQL DDL files.");
158             }
159         }
160 
161         //Find the entity classes in the packages.
162         final Set<Class<?>> entityClasses = new HashSet<>();
163         for (final String packageName : packages) {
164             final Set<Class<?>> packageEntities = EntityFinder
165                 .forPackage(project, getLog(), packageName, includeTestClasses)
166                 .findEntities();
167             entityClasses.addAll(packageEntities);
168         }
169         getLog().info(String.format("Found %d entities.",
170                                     entityClasses.size()));
171 
172         // Find the DDL generator implementation to use.
173         final ServiceLoader<DdlGenerator> serviceLoader = ServiceLoader
174             .load(DdlGenerator.class);
175         final DdlGenerator ddlGenerator;
176         if (serviceLoader.iterator().hasNext()) {
177             ddlGenerator = serviceLoader.iterator().next();
178         } else {
179             throw new MojoFailureException(String.format(
180                 "No implementation of '%s' is available.",
181                 DdlGenerator.class.getName()));
182         }
183 
184         for (final Dialect dialect : convertDialects()) {
185             ddlGenerator.generateDdl(dialect, entityClasses, this);
186         }
187 
188         if (customDialects != null) {
189             for (final String customDialect : customDialects) {
190                 ddlGenerator.generateDdl(customDialect, entityClasses, this);
191             }
192         }
193     }

301     /**
302      * Helper method which reads the dialects from the parameter and converts
303      * them into instances of the {@link Dialect} enumeration.
304      *
305      * @return A list of all dialects to use
306      *
307      * @throws MojoFailureException If an error occurs.
308      */
309     private Set<Dialect> convertDialects() throws MojoFailureException {
310 
311         final Set<Dialect> dialectsList = new HashSet<>();
312         if (dialects != null) {
313             for (final String dialect : dialects) {
314                 convertDialect(dialect, dialectsList);
315             }
316         }
317         return dialectsList;
318     }
319 
320     /**
321      * Helper method for converting the dialects from {@code String} to
322      * instances of the {@link Dialect} enumeration.
323      *
324      * @param dialect      The dialect to convert.
325      * @param dialectsList The lists of dialects where the converted dialect is
326      *                     stored.
327      *
328      * @throws MojoFailureException If the dialect string could not be
329      *                              converted, for example if it is misspelled.
330      *                              This will cause a {@code Build Failure}
331      */
332     private void convertDialect(final String dialect,
333                                 final Set<Dialect> dialectsList)
334         throws MojoFailureException {
335 
336         try {
337             dialectsList.add(Dialect
338                 .valueOf(dialect.toUpperCase(Locale.ENGLISH)));
339         } catch (IllegalArgumentException ex) {
340             final StringBuffer buffer = new StringBuffer();
341             for (final Dialect avilable : Dialect.values()) {
342                 buffer.append(avilable.toString()).append('\n');
343             }
344 
345             throw new MojoFailureException(
346                 String.format(
347                     "Can't convert the configured dialect '%s' to a dialect classname. "
348                     + "Available dialects are:%n"
349                         + "%s",
350                     dialect,
351                     buffer.toString()),
352                 ex);
353         }
354     }

356     protected void writeOutputFile(final String dialectClassName,
357                                    final Path tmpDir)
358         throws MojoFailureException {
359 
360         final OutputFileWriter writer = new OutputFileWriter(outputDirectory);
361         writer.setOmitDialectFromFileName(omitDialectFromFileName
362                                               && dialects.length == 1);
363         writer.setOutputFileNamePrefix(outputFileNamePrefix);
364         writer.setOutputFileNameSuffix(outputFileNameSuffix);
365 
366         writer.writeOutputFile(dialectClassName, tmpDir);
367 
368     }

370     public String getDialectNameFromClassName(final String dialectClassName) {
371 
372         final int pos = dialectClassName.lastIndexOf('.');
373 
374         if (dialectClassName.toLowerCase(Locale.ROOT).endsWith("dialect")) {
375             return dialectClassName
376                 .substring(pos + 1,
377                            dialectClassName.length() - "dialect".length())
378                 .toLowerCase(Locale.ROOT);
379         } else {
380             return dialectClassName.substring(pos + 1)
381                 .toLowerCase(Locale.ROOT);
382         }
383     }
